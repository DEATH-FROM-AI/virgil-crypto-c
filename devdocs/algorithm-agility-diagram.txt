@startuml

enum AlgId {
    none
    sha224
    sha256
    sha384
    sha512
    kdf1
    kdf2
}
'
interface AlgInfo {
    algId(): AlgId
}

class SimpleAlgInfo

class KdfAlgInfo {
    + hashAlgInfo : SimpleAlgInfo
    + salt : Byte[]
    + iterationCount : Integer
}

interface Alg {
    algId(): AlgId
    produceAlgInfo() : AlgInfo
    restoreAlgInfo(: AlgInfo)
}

interface AlgInfoSerializer {
    serialize(: AlgInfo) : Byte[]
}

interface AlgInfoDeserializer {
    deserialize(: Byte[]) : AlgInfo
}

class AlgInfoDerSerializer
class AlgInfoDerDeserializer
class SimpleAlgInfoDerSerializer
class SimpleAlgInfoDerDeserializer
class KdfAlgInfoDerSerializer
class KdfAlgInfoDerDeserializer


AlgInfo .left.> AlgId : <<use>>
SimpleAlgInfo .up.|> AlgInfo
KdfAlgInfo .up.|> AlgInfo
Alg .left.> AlgInfo : <<produce/consume>>

AlgInfoDerSerializer .down.|> AlgInfoSerializer
AlgInfoDerDeserializer .down.|> AlgInfoDeserializer
SimpleAlgInfoDerSerializer .down.|> AlgInfoSerializer
SimpleAlgInfoDerDeserializer .down.|> AlgInfoDeserializer
KdfAlgInfoDerSerializer .down.|> AlgInfoSerializer
KdfAlgInfoDerDeserializer .down.|> AlgInfoDeserializer
AlgInfoSerializer .down.> AlgInfo : <<produce/consume>>
AlgInfoDeserializer .down.> AlgInfo : <<produce/consume>>

class sha256
class kdf1

sha256 -up-|> Alg
kdf1  -up-|> Alg

note bottom of sha256
    <<produce/consume>>
    SimpleAlgInfo
end note

note bottom of kdf1
    <<produce/consume>>
    KdfAlgInfo
end note

@enduml
